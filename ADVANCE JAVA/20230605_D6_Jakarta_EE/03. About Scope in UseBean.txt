<jsp:useBean id="ob1" class="beanpack.MyClass"/>
	is same as
<%
pageContext.setAttribute("ob1",new beanpack.MyClass());
%>


<jsp:useBean id="ob1" scope="session" class="beanpack.MyClass"/>

	is same as

<%
session.setAttribute("ob1",new beanpack.MyClass());
%>
