<bean id="advisor1" 
                 class="org.springframework.aop.support.DefaultPointcutAdvisor">
		<property name="pointcut" ref="pointcut1" />
		<property name="advice" ref="afterCall"/>
</bean>

		above tag means
			advisor1 -  "after" advice -   "pointcut1" ( multiply )

<bean id="advisor2" 
                 class="org.springframework.aop.support.DefaultPointcutAdvisor">
		<property name="pointcut" ref="pointcut2" />
		<property name="advice" ref="beforeCall"/>
</bean>

		above tag means
			advisor2 - "before"   - pointcut2 ( all the methods)
