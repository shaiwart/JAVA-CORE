
				Without AOP
// in case of following example cross-cutting concerns are tightly coupled with business logic.

class Account
{
	void deposit(double amt)	
	{
		// deposit logic    business logic
		// transaction	cross-cutting concern
		// logging	cross-cutting concern
	}
	double withdraw()
	{
		// withdraw logic   business logic
		//transaction	cross-cutting concern
		//logging	cross-cutting concern
		//security	cross-cutting concern
	}
}








			without AOP

// in case of following example cross-cutting concerns are defined separately (loosely coupled with business logic i.e. deposit and withdraw) but since they are invoked from deposit and withdraw methods still there is some kind of tight coupling exists.

class Account
{
	void deposit(double amt)	
	{
		// deposit logic    business logic
		invoke transaction
		invoke logging	
	}
	double withdraw()
	{
		// withdraw logic   business logic
		invoke transaction
		invoke logging	
		invoke security	
	}
}

class Cross_Cut
{
	void transaction()
	{
		// code for transaction
	}
	void logging()
	{
		// code for logging
	}
	void security()
	{
		// code for security
	}
}




				



With AOP

	class Account
	{
		void deposit(double amt)	
		{
			// deposit logic    business logic
		}
		double withdraw()
		{
			// withdraw logic   business logic
		}
	}

		// cross cutting concerns

	class transaction
	{
		transaction_logic()		
		{
			code for transaction
		}
	}
	class security
	{
		security_logic()
		{
			code for security
		}
	}
	class logging
	{
		logging_logic()
		{
			code for logging
		}
	}




inside Spring bean configuration file developer has to mention:
	which cross cutting concern/s should be applied with which business logic method.
	e.g
		apply "logging" and "transaction" with "deposit"
		apply "logging","transaction" and "security"  with "withdraw"


and then during runtime:

	}
